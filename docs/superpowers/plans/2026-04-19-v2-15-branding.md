> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the Grid Shield logo, correct SVG assets, favicon bundle, PWA manifest, dynamic favicon state, and OG image per Part 5 of backupos-ui-spec-v2.md.

**Architecture:** SVG assets live in `apps/web/public/`. PNG icons are generated via Next.js built-in `icon.tsx` / `apple-icon.tsx` routes (no external deps — uses `ImageResponse` from `next/og` which ships with Next.js 15). The sidebar `Logo` component gets corrected to match spec dimensions. A `FaviconManager` client component polls a server action every 30 s and swaps `<link rel="icon">` href between three static SVG variants (idle/running/alert) while updating `document.title`.

**Tech Stack:** Next.js 15 App Router, `next/og` ImageResponse, TypeScript strict, SVG, Drizzle ORM for status polling.

- [ ] **Step 1: Create `apps/web/app/icon.tsx`**

This renders the Grid Shield at two sizes via `generateImageMetadata`. The scale factor from 48-viewBox to pixel sizes: for 192px → factor=4; for 512px → factor≈10.67.

```typescript
import { ImageResponse } from 'next/og'

export const contentType = 'image/png'

export function generateImageMetadata() {
  return [
    { id: '192', size: { width: 192, height: 192 }, contentType },
    { id: '512', size: { width: 512, height: 512 }, contentType },
  ]
}

function GridShield({ size }: { size: number }) {
  const f = size / 48
  const r = (v: number) => Math.round(v * f)
  return (
    <div style={{
      width: size, height: size, background: '#1A1206',
      borderRadius: r(16), display: 'flex', position: 'relative', overflow: 'hidden',
    }}>
      <div style={{ position: 'absolute', left: r(6),  top: r(6),  width: r(16), height: r(16), background: '#F5A623', borderRadius: r(2) }} />
      <div style={{ position: 'absolute', left: r(26), top: r(6),  width: r(16), height: r(16), background: '#854F0B', borderRadius: r(2) }} />
      <div style={{ position: 'absolute', left: r(6),  top: r(26), width: r(16), height: r(16), background: '#854F0B', borderRadius: r(2) }} />
      <div style={{ position: 'absolute', left: r(26), top: r(26), width: r(16), height: r(16), background: '#C77A14', borderRadius: r(2) }} />
```

```typescript
'use server'

import { getDb, backupRuns, storageAlerts } from '@backupos/db'
import { eq, sql }                           from 'drizzle-orm'

export interface SystemStatus {
  activeRunCount: number
  alertCount:     number
}

export async function getSystemStatus(): Promise<SystemStatus> {
  const db = getDb()

  const [runRow] = db
    .select({ n: sql<number>`count(*)` })
    .from(backupRuns)
    .where(eq(backupRuns.status, 'running'))
    .all()

  const [alertRow] = db
    .select({ n: sql<number>`count(*)` })
    .from(storageAlerts)
    .all()

  return {
    activeRunCount: runRow?.n ?? 0,
    alertCount:     alertRow?.n ?? 0,
  }
}
```

- [ ] **Step 3: Create `apps/web/components/favicon-manager.tsx`**

```typescript
'use client'

import { useEffect, useRef } from 'react'
import { getSystemStatus }   from '@/app/actions/system-status'

const FAVICONS = {
  idle:    '/favicon.svg',
  running: '/favicon-running.svg',
  alert:   '/favicon-alert.svg',
} as const

function setFavicon(href: string) {
  let link = document.querySelector<HTMLLinkElement>('link[rel="icon"]')
  if (!link) {
    link = document.createElement('link')
    link.rel = 'icon'
    document.head.appendChild(link)
  }
  link.href = href
}

function setTitle(activeRuns: number, alerts: number) {
  const base = 'BackupOS'
  if (alerts > 0) {
    document.title = `(${alerts}) ${base}`
  } else if (activeRuns > 0) {
    document.title = `● ${base}`
  } else {
    document.title = base
  }
}

export function FaviconManager() {
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null)

  async function poll() {
    try {
      const { activeRunCount, alertCount } = await getSystemStatus()
      if (alertCount > 0) {
        setFavicon(FAVICONS.alert)
      } else if (activeRunCount > 0) {
        setFavicon(FAVICONS.running)
      } else {
        setFavicon(FAVICONS.idle)
      }
      setTitle(activeRunCount, alertCount)
    } catch {
      // silently ignore — favicon state is non-critical
    }
  }

  useEffect(() => {
    poll()
    intervalRef.current = setInterval(poll, 30_000)
    return () => {
      if (intervalRef.current) clearInterval(intervalRef.current)
    }
  }, [])

  return null
}
```
